# Base32 Crockford

## Install
`npm i --save @dwlib/base32-crockford`

## Usage
```javascript
// CJS
const Base32Crockford = require('@dwlib/base32-crockford');
const Base32CrockfordDecode = require('@dwlib/base32-crockford/decode');
const Base32CrockfordDecodeBigInt = require('@dwlib/base32-crockford/decodeBigInt');
const Base32CrockfordDecodeInt = require('@dwlib/base32-crockford/decodeInt');
const Base32CrockfordDecodeInto = require('@dwlib/base32-crockford/decodeInto');
const Base32CrockfordEncode = require('@dwlib/base32-crockford/encode');
const Base32CrockfordEncodeBigInt = require('@dwlib/base32-crockford/encodeBigInt');
const Base32CrockfordEncodeInt = require('@dwlib/base32-crockford/encodeInt');
const Base32CrockfordIsValid = require('@dwlib/base32-crockford/isValid');
const Base32CrockfordNormalize = require('@dwlib/base32-crockford/normalize');
const Base32CrockfordValidate = require('@dwlib/base32-crockford/validate');
// ESM
import Base32Crockford from '@dwlib/base32-crockford';
import Base32CrockfordDecode from '@dwlib/base32-crockford/decode';
import Base32CrockfordDecodeBigInt from '@dwlib/base32-crockford/decodeBigInt';
import Base32CrockfordDecodeInt from '@dwlib/base32-crockford/decodeInt';
import Base32CrockfordDecodeInto from '@dwlib/base32-crockford/decodeInto';
import Base32CrockfordEncode from '@dwlib/base32-crockford/encode';
import Base32CrockfordEncodeBigInt from '@dwlib/base32-crockford/encodeBigInt';
import Base32CrockfordEncodeInt from '@dwlib/base32-crockford/encodeInt';
import Base32CrockfordIsValid from '@dwlib/base32-crockford/isValid';
import Base32CrockfordNormalize from '@dwlib/base32-crockford/normalize';
import Base32CrockfordValidate from '@dwlib/base32-crockford/validate';
```
